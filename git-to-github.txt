# 1. 初始化 git (如果还没初始化)
git init

# 2. 添加所有文件到暂存区
git add .

# 3. 提交到本地仓库
git commit -m "Initial commit"

# 4. 重命名默认分支为 main (GitHub 现在默认使用 main)
git branch -M main

# 5. 关联远程仓库 (替换为你自己的仓库地址)
git remote add origin https://github.com/你的用户名/你的仓库名.git

# 6. 推送代码 (-u 设置上游分支，以后只需输 git push)
git push -u origin main
